POCET OHODNOTENI PRE JEDNNOTLIVE SLOVA
--------------------------------------------------------------
SELECT lemma,COUNT(zo.id) as pocet FROM sk_lemmas_synonyms
JOIN zbieranie_ohodnoteni zo
ON lemma=zo.prve_slovo
WHERE pos LIKE 'A'
GROUP BY lemma
ORDER BY pocet ASC


VYTBORENIE KOMPLET SLOVNIKA
---------------------
SELECT prve_slovo, druhe_slovo, COUNT(druhe_slovo) as amount, SUM(vzdialenost) * 1.0 / COUNT(druhe_slovo) * 1.0 as coefficient
FROM zbieranie_ohodnoteni
GROUP BY prve_slovo, druhe_slovo
HAVING SUM(vzdialenost) > 0
ORDER BY prve_slovo ASC, coefficient DESC, amount ASC


VYTVORENIE SLOVNIKA LEN PRE Ackove
--------------------------------------
SELECT prve_slovo, druhe_slovo, COUNT(druhe_slovo) as amount, SUM(vzdialenost) * 1.0 / COUNT(druhe_slovo) * 1.0 as coefficient
FROM zbieranie_ohodnoteni
JOIN sk_lemmas_synonyms sls
ON prve_slovo = sls.lemma
WHERE sls.pos LIKE 'A'
GROUP BY prve_slovo, druhe_slovo
HAVING SUM(vzdialenost) > 0
ORDER BY prve_slovo ASC, coefficient DESC, amount ASC
